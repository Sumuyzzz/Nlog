```javascript
function rev(arr, n) {
    for (let i = 0;i < n;i++) {
        arr.unshift(arr.pop());
    }
    return arr;

}

result = rev([1, 2, 3, 4, 5, 6, 7, 8, 9], 6)
console.log(result)
//[4, 5, 6, 7, 8, 9, 1, 2, 3]
```

最大值

```javascript
function max(array) {
    let element = array[0];
    for (let i = 0;i < array.length;i++) {
        if (array[i] > element) {
            element = array[i];
        }
    }
    return element
}

let result = max([5, 8, 9, 31, 2, 678, 7])
console.log(result)
//678
```

```javascript
//indexOf方法:获取到数组当中某一个元素索引[没有则输出-1]
function unRepeat(array) {
    let result = []
    for (let i = 0;i < array.length;i++) {
        if (result.indexOf(array[i]) == -1) {
            result.push(array[i])
        }

    }
    return result
}
let result1 = unRepeat([1, 1, 2, 2, 3, 1, 2, 3, 5, 7, 8, 5])
console.log(result1)
//[ 1, 2, 3, 5, 7, 8 ]
```

### 2 字符串的实战与练习

```javascript
//问题:封装一个函数,函数功能是可以就爱那个是餐传递进来的字符进行倒置返回
function reString(str) {
    //累加器
    let output = ''
    //反向循环体
    for (let i = str.length - 1;i >= 0;i--) {
        output += str[i]
    }
    console.log(output)
}
reString('你干嘛啦')
```

```javascript
//方法二
function reString(str) {
    //split:可以将字符切割为数组
    let arr = str.split('')
    //reverse:可以将数组元素进行倒置
    arr.reverse()
    //join:可以将数组转换为字符串
    let result = arr.join('')
    //返回结果
    return result
}
//使用函数
let output = reString('我爱你北京天安门')
console.log(output)
```

```javascript
//大小写转换
function changesString(str) {
    //累加器
    let output = ''
    for (let index = 0;index < str.length;index++) {
        //获取所有的大写字母
        if (str[index] < 'a') {
            //进入这个条件一定是大写
            output += str[index].toLowerCase();
        } else {
            //进入这个条件一定是小写
            output += str[index].toUpperCase();
        }
    }
    return output;
}
let newStr = changesString('i lIKE mILK')
console.log(newStr)
//I Like Milk
```
