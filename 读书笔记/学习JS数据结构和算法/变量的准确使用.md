```javascript
let movie = 'Lord of the Rings'; // 使用let声明一个指环王
// var movie = 'Batman v Superman'; //throws error, variable movie already declared/

function starWarsFan() {
  const movie = 'Star Wars'; //  使用const更改movie为一个星球大战
  return movie;
}

function marvelFan() {
  movie = 'The Avengers'; // 不使用任何关键字声明一个复仇者(不赞同)
  return movie;
}

function blizzardFan() {
  const isFan = true;
  let phrase = 'Warcraft'; // 用let声明一个魔兽世界
  console.log('Before if: ' + phrase);//前面打印
  if (isFan) {
    let phrase = 'initial text'; // 初始化文本
    phrase = 'For the Horde!'; // 更改短语为'为了部落'
    console.log('Inside if: ' + phrase);在两者之间打印
  }
  phrase = 'For the Alliance!'; // 否则短语更改为'为了联盟'
  console.log('After if: ' + phrase);后面打印
}

console.log(movie); // Lord of the Rings
console.log(starWarsFan()); // Star Wars 这里
console.log(marvelFan()); // The Avengers
console.log(movie); // {11}
blizzardFan(); // {12}

// output
// Lord of the Rings
// Star Wars
// The Avengers
// The Avengers
// Before if: Warcraft
// Inside if: For the Horde!
// After if: For the Alliance!

```
