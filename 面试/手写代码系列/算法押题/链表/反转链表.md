var reverse = (head) => {

 if (!head || !head.next) return head;

 // 从后往前翻

 const pre = reverse(head.next);

 head.next = pre.next;

 pre.next = head;

 return head;

}

  
  
  

var reverseList = function (head) {

 let cur = head;

 while (cur && cur.next) {

 cur = cur.next;

 }

 reverse(head);

 return cur;

  

};
