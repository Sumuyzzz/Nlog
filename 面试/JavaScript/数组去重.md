## 使用set实现
```js
function unique(arr){
	return Array.form(new Set(arr))
}
```

用for嵌套实现
```js

function unique(arr){
	for(let i=0;i<arr.length;i++){
		for(let j=i+1;j<arr.length;j++){
		if(arr[i]=arr[j]){
			arr.slice(j,1);
			j--;
			}
		}
	}
	return arr
}
```
